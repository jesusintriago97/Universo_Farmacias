

	/* ¿Cuáles son los empleados mas antigos dentro de la cadena de negocios?*/                                          */
	/*==============================================================*/

SELECT 
	VENDEDOR.NOMBRE_VENDEDOR AS NOMBRE,
	VENDEDOR.APELLIDOS__VENDEDOR AS APELLIDOS,
	VENDEDOR.FECHA_INGRESO AS FECHA_INGRESO,
	VENDEDOR.DIRECCION_VENDEDOR AS DIRECCION
FROM SUCURSAL INNER JOIN VENDEDOR ON SUCURSAL.ID_SUCURSAL = VENDEDOR.ID_SUCURSAL
WHERE VENDEDOR.ID_ESTADO = 1 
	GROUP BY 
	  FECHA_INGRESO,
	  NOMBRE, 
	  APELLIDOS,
	  DIRECCION
	  LIMIT 4
;

SELECT 
	PRODUCTO.NOMBRE_PRODUCTO AS NOMBRE_PRODUCTO,
	DETALLE_DE_LA_VENTA.CANTIDAD_VENTA AS CANTIDAD_PRODUCTO,
	COUNT (DETALLE_DE_LA_VENTA.CANTIDAD_VENTA)
FROM VENTA INNER JOIN DETALLE_DE_LA_VENTA ON VENTA.ID_VENTA = DETALLE_DE_LA_VENTA.ID_VENTA
		   INNER JOIN PRODUCTO ON PRODUCTO.ID_PRODUCTO = DETALLE_DE_LA_VENTA.ID_PRODUCTO
		   INNER JOIN VENDEDOR ON VENDEDOR.ID_VENDEDOR = VENTA.ID_VENDEDOR
		   INNER JOIN SUCURSAL ON SUCURSAL.ID_SUCURSAL = VENDEDOR.ID_SUCURSAL
WHERE SUCURSAL.ID_SUCURSAL =3
	GROUP BY
	  NOMBRE_PRODUCTO,
	  CANTIDAD_PRODUCTO
	ORDER BY
	  NOMBRE_PRODUCTO aSC
LIMIT 1;

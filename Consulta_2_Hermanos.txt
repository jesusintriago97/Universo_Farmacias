
//¿Hay hermanos trabajando dentro de una sucursal?//

SELECT 
	VENDEDOR.NOMBRE_VENDEDOR AS NOMBRE,
	VENDEDOR.APELLIDOS__VENDEDOR AS APELLIDOS,
	SUCURSAL.DIRECCION_SUCURSAL AS DIRECCION,
	VENDEDOR.DIRECCION_VENDEDOR AS DIRECCION_VENDEDOR,
	COUNT (VENDEDOR.NOMBRE_VENDEDOR)
FROM ESTADO_DEL_VENDEDOR INNER JOIN VENDEDOR ON ESTADO_DEL_VENDEDOR.ID_ESTADO = VENDEDOR.ID_ESTADO
	INNER JOIN  SUCURSAL ON SUCURSAL.ID_SUCURSAL = VENDEDOR.ID_SUCURSAL
WHERE VENDEDOR.APELLIDOS__VENDEDOR = 'Bailon Intriago' AND SUCURSAL.ID_SUCURSAL =1
	GROUP BY
	  Nombre_vendedor,
	  Apellidos__vendedor,
	  direccion_sucursal,
	  direccion_vendedor;


//¿Hay hermanos trabajando dentro de la cadena del negocio que estén distintas sucursales?//



SELECT 
	VENDEDOR.NOMBRE_VENDEDOR AS NOMBRE,
	VENDEDOR.APELLIDOS__VENDEDOR AS APELLIDOS,
	SUCURSAL.DIRECCION_SUCURSAL AS DIRECCION,
	VENDEDOR.DIRECCION_VENDEDOR AS DIRECCION_VENDEDOR,
	SUCURSAL.ID_SUCURSAL AS ID_SUCUR,
	
	COUNT (VENDEDOR.NOMBRE_VENDEDOR)
FROM ESTADO_DEL_VENDEDOR INNER JOIN VENDEDOR ON ESTADO_DEL_VENDEDOR.ID_ESTADO = VENDEDOR.ID_ESTADO
	INNER JOIN  SUCURSAL ON SUCURSAL.ID_SUCURSAL = VENDEDOR.ID_SUCURSAL
WHERE VENDEDOR.APELLIDOS__VENDEDOR = 'Bailon Intriago' 
	GROUP BY
	  Nombre_vendedor,
	  Apellidos__vendedor,
	  direccion_sucursal,
	  direccion_vendedor,
	  Id_sucur;

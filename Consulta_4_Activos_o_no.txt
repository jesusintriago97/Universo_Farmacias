/*     ¿Qué cantidad de esas personas continuaron trabajando en la sucursal?      */


SELECT 
	VENDEDOR.NOMBRE_VENDEDOR AS NOMBRE,
	VENDEDOR.APELLIDOS__VENDEDOR AS APELLIDOS,
	SUCURSAL.DIRECCION_SUCURSAL AS DIRECCION,
	VENDEDOR.ID_ESTADO AS ESTAD,
	SUCURSAL.ID_SUCURSAL AS ID_SUCUR,
	
	
	
	COUNT (VENDEDOR.NOMBRE_VENDEDOR)
FROM ESTADO_DEL_VENDEDOR INNER JOIN VENDEDOR ON ESTADO_DEL_VENDEDOR.ID_ESTADO = VENDEDOR.ID_ESTADO
	INNER JOIN  SUCURSAL ON SUCURSAL.ID_SUCURSAL = VENDEDOR.ID_SUCURSAL
WHERE ESTADO_DEL_VENDEDOR.ID_ESTADO = 1
	GROUP BY
	  Nombre_vendedor,
	  Apellidos__vendedor,
	  direccion_sucursal,
	  ESTAD,
	  Id_sucur;




/*¿Qué cantidad de esas personas fueron retiradas del negocio?    */

 
SELECT 
	VENDEDOR.NOMBRE_VENDEDOR AS NOMBRE,
	VENDEDOR.APELLIDOS__VENDEDOR AS APELLIDOS,
	SUCURSAL.DIRECCION_SUCURSAL AS DIRECCION,
	VENDEDOR.ID_ESTADO AS ESTAD,
	SUCURSAL.ID_SUCURSAL AS ID_SUCUR,
	
	
	
	COUNT (VENDEDOR.NOMBRE_VENDEDOR)
FROM ESTADO_DEL_VENDEDOR INNER JOIN VENDEDOR ON ESTADO_DEL_VENDEDOR.ID_ESTADO = VENDEDOR.ID_ESTADO
	INNER JOIN  SUCURSAL ON SUCURSAL.ID_SUCURSAL = VENDEDOR.ID_SUCURSAL
WHERE ESTADO_DEL_VENDEDOR.ID_ESTADO = 2
	GROUP BY
	  Nombre_vendedor,
	  Apellidos__vendedor,
	  direccion_sucursal,
	  ESTAD,
	  Id_sucur;
